// La URL de tu script ahora es una variable de entorno, m√°s segura.
const SPREADSHEET_WEB_APP_URL = process.env.GOOGLE_SHEET_URL;

export default async function handler(req: any, res: any) {
    if (req.method !== 'POST') {
        return res.status(405).json({ error: 'Method Not Allowed' });
    }

    if (!SPREADSHEET_WEB_APP_URL) {
        console.warn('Google Sheets URL no configurada en el servidor.');
        // No devolvemos error al cliente, simplemente no guardamos.
        return res.status(200).json({ message: 'OK' });
    }

    const { email } = req.body;
    if (!email) {
        return res.status(400).json({ error: 'Email is required' });
    }

    try {
        await fetch(SPREADSHEET_WEB_APP_URL, {
            method: 'POST',
            mode: 'cors',
            headers: {
                'Content-Type': 'application/json',
            },
            body: JSON.stringify({ email }),
        });
        
        return res.status(200).json({ message: 'Email saved' });

    } catch (error) {
        console.error('Error al contactar Google Sheets:', error);
        return res.status(500).json({ error: 'Failed to save email' });
    }
}
